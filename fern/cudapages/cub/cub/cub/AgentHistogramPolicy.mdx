Parameterizable tuning policy type for AgentHistogram.

<AccordionGroup>
<Accordion title="Template parameters">

<ParamField path="BlockThreads" type="int">
Threads per thread block
</ParamField>

<ParamField path="PixelsPerThread" type="int">
Pixels per thread (per tile of input)
</ParamField>

<ParamField path="LoadAlgorithm" type="BlockLoadAlgorithm">
The [BlockLoad](/library/api/cub::_block_load) algorithm to use
</ParamField>

<ParamField path="LoadModifier" type="CacheLoadModifier">
Cache load modifier for reading input elements
</ParamField>

<ParamField path="RleCompress" type="bool">
Whether to perform localized RLE to compress samples before histogramming
</ParamField>

<ParamField path="MemoryPreference" type="BlockHistogramMemoryPreference">
Whether to prefer privatized shared-memory bins (versus privatized global-memory bins)
</ParamField>

<ParamField path="WorkStealing" type="bool">
Whether to dequeue tiles from a global work queue
</ParamField>

<ParamField path="VecSize" type="int" default="4">
Vector size for samples loading (1, 2, 4)
</ParamField>

</Accordion>
